#!/bin/bash
Year=`date +%Y`
Month=`date +%m`
Day=`date +%d`
Hour=`date +%H`
Minute=`date +%M`
Second=`date +%S`
echo `date`
echo "Current Date is: $Day-$Month-$Year"
echo "Current Time is: $Hour:$Minute:$Second"

OUTPUT-
Fri 19 Nov 13:54:34 UTC 2021
Current Date is: 19-11-2021
Current Time is: 13:54:34
